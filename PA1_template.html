5. Are there differences in activity patterns between weekdays and weekends?

weekDay<-function(dt)
{
        s = weekdays(dt)
        for(i in 1:length(dt))
        {
                if(s[i] == "Saturday" | s[i] == "Sunday")
                        s[i]="weekend"
                else
                        s[i]="weekday"

        }
        s
}
