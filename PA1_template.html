Are there differences in activity patterns between weekdays and weekends?

weekDay<-function(dt)
{
        s = weekdays(dt)
        for(i in 1:length(dt))
        {
                if(s[i] == "Saturday" | s[i] == "Sunday")
                        s[i]="weekend"
                else
                        s[i]="weekday"

        }
        s
}

newdata<- mutate(newdata, WD = weekDay(date))
library(lattice)
newdata<-group_by(newdata, interval, WD)
newdata<-summarise(newdata, avg = mean(steps))
xyplot(newdata$avg ~ newdata$interval| newdata$WD, type = "l")
